## C：并查集

判断环的个数即可

```C++
#include <bits/stdc++.h>
//# pragma GCC optimize(3)
#define int long long
#define endl "\n"
using namespace std;
// 其实就i是找环2 个数cnt  c在环中则删去  ans=2^cnt
const int N = 2e5 + 5;
const int mod=1e9+7;
int T, n, a[N],b[N],c[N];
int pre[N];

int qpow(int x,int k){
    int res=1;
    while(k){
        if(k&1)res=res*x%mod;
        x=x*x%mod;
        k>>=1;
    }
    return res%mod;
}
int find(int x){
    if(x!=pre[x])pre[x]=find(pre[x]);
    return pre[x];
}
int vis[N],vis2[N];
void solve(){
    cin>>n;
for(int i=1;i<=n;i++)vis[i]=0,vis2[i]=0;

    for(int i=1;i<=n;i++)pre[i]=i;
    for(int i=1;i<=n;i++)cin>>a[i];
    for(int i=1;i<=n;i++)cin>>b[i];
    for(int i=1;i<=n;i++)cin>>c[i];
    for(int i=1;i<=n;i++){
        if(a[i]==b[i])vis2[i]=1; //一个点的集合不算环
        int x=find(a[i]),y=find(b[i]); 
            if(x!=y)pre[x]=y;            //一个点不算  
    }

    int ans=1;
    int cnt=0;
    for(int i=1;i<=n;i++){
        if(find(i)==i)cnt++;  // 集合个数
        if(vis2[i])cnt--;
    }
    for(int i=1;i<=n;i++){ //环中有c 对应的数则删去
        if(c[i]&&!vis2[i]){
        int x=find(c[i]);
        if(!vis[x]){
            vis[x]=1;
            cnt--;
        }
        }
    }
    ans=qpow(2,cnt);
    if(n==1)ans=1;
    cout<<ans<<endl;
}
signed main() {
  std::ios::sync_with_stdio(false);
  cin.tie(0);
  cout.tie(0);

  cin >> T;
  while (T--) {
      solve();
  }
  return 0;
}

```
